1//! Structured logging for VRF operations and audit trails
2//!
3//! This module provides comprehensive logging capabilities for production VRF deployments,
4//! including audit trails, performance monitoring, and debugging support.
5//!
6//! # Features
7//!
8//! - **Structured Logging**: JSON and text formats for easy parsing
9//! - **Audit Trails**: Comprehensive operation tracking for compliance
10//! - **Performance Monitoring**: Duration tracking for all operations
11//! - **Security Logging**: Key usage tracking without exposing sensitive data
12//! - **Debug Support**: Detailed diagnostics with `vrf-debug` feature
13//!
14//! # Usage
15//!
16//! ```rust
17//! use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
18//!
19//! let logger = VrfLogger::new(LogLevel::Info);
20//!
21//! // Log a successful operation
22//! logger.info(VrfOperation::Prove, "VRF proof generated successfully".to_string());
23//!
24//! // Log an error
25//! logger.error(VrfOperation::Verify, "Proof verification failed".to_string());
26//!
27//! // Log with key identifier (doesn't expose key material)
28//! logger.info(
29//!     VrfOperation::KeyGeneration,
30//!     format!("Generated key for validator-001")
31//! );
32//! ```
33//!
34//! # Security Considerations
35//!
36//! The logging system is designed with security in mind:
37//! - Never logs private key material
38//! - Public keys and key IDs are logged only when necessary
39//! - Timing information helps detect side-channel attacks
40//! - All log entries include timestamps for audit trails
41//!
42//! # Output Formats
43//!
44//! ## Text Format (Human-Readable)
45//!
46//! ```text
47//! [2025-11-09 10:30:45] INFO PROVE VRF proof generated successfully
48//! [2025-11-09 10:30:46] ERROR VERIFY Proof verification failed
49//! ```
50//!
51//! ## JSON Format (Machine-Readable)
52//!
53//! ```json
54//! {
55//!   "timestamp": "2025-11-09T10:30:45Z",
56//!   "level": "INFO",
57//!   "operation": "PROVE",
58//!   "message": "VRF proof generated successfully",
59//!   "success": true
60//! }
61//! ```

65/// Log level for VRF operations
66///
67/// Defines the severity levels for log messages, allowing filtering
68/// of log output based on importance and detail level.
69///
70/// # Order
71///
72/// Log levels are ordered from most verbose (Debug) to least verbose (Error):
73/// `Debug < Info < Warning < Error`
74///
75/// # Examples
76///
77/// ```rust
78/// use cardano_vrf::LogLevel;
79///
80/// let level = LogLevel::Info;
81/// assert!(level > LogLevel::Debug);
82/// assert!(level < LogLevel::Error);
83/// ```
84#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
85pub enum LogLevel {
86    /// Debug-level logging for detailed diagnostics
87    ///
88    /// Use for development and troubleshooting. Includes detailed
89    /// information about internal operations and state.
90    Debug,
91
92    /// Informational messages about normal operations
93    ///
94    /// Standard operational messages indicating successful completion
95    /// of routine tasks. Suitable for production monitoring.
96    Info,
97
98    /// Warning messages about potential issues
99    ///
100    /// Non-critical issues that don't prevent operation but may
101    /// require attention or indicate suboptimal conditions.
102    Warning,
103
104    /// Error messages for operation failures
105    ///
106    /// Critical failures that prevent successful operation completion.
107    /// Requires immediate attention in production environments.
108    Error,
109}
110
111impl fmt::Display for LogLevel {
112    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
113        match self {
114            LogLevel::Debug => write!(f, "DEBUG"),
115            LogLevel::Info => write!(f, "INFO"),
116            LogLevel::Warning => write!(f, "WARN"),
117            LogLevel::Error => write!(f, "ERROR"),
118        }
119    }
120}

122/// VRF operation types for logging and categorization
123///
124/// Categorizes different types of VRF operations for structured logging,
125/// metrics collection, and audit trails. Each operation type represents
126/// a distinct cryptographic or key management activity.
127///
128/// # Usage
129///
130/// Operation types are used with logging and metrics to track:
131/// - Operation frequency and patterns
132/// - Performance characteristics per operation
133/// - Security-relevant events
134/// - Failure modes and error rates
135///
136/// # Examples
137///
138/// ```rust
139/// use cardano_vrf::{VrfOperation, VrfLogger, LogLevel};
140///
141/// let logger = VrfLogger::new(LogLevel::Info);
142///
143/// // Log different operation types
144/// logger.info(VrfOperation::Prove, "Generated proof".to_string());
145/// logger.info(VrfOperation::Verify, "Verified proof".to_string());
146/// logger.info(VrfOperation::KeyGeneration, "Created new key".to_string());
147/// ```
148#[derive(Debug, Clone, Copy)]
149pub enum VrfOperation {
150    /// VRF proof generation operation
151    ///
152    /// Logged when generating a VRF proof from a secret key and message.
153    /// Performance-critical operation in block production scenarios.
154    Prove,
155
156    /// VRF proof verification operation
157    ///
158    /// Logged when verifying a VRF proof against a public key and message.
159    /// Common operation during blockchain validation.
160    Verify,
161
162    /// VRF keypair generation operation
163    ///
164    /// Logged when creating new VRF keypairs. Security-sensitive operation
165    /// that should be audited in production environments.
166    KeyGeneration,
167
168    /// VRF key retrieval from storage operation
169    ///
170    /// Logged when loading existing keys from storage or HSM.
171    /// Useful for tracking key access patterns.
172    KeyRetrieval,
173
174    /// Hardware Security Module (HSM) operation
175    ///
176    /// Logged for HSM-specific operations including key management,
177    /// signing, and device communication.
178    HsmOperation,
179}
180
181impl fmt::Display for VrfOperation {
182    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
183        match self {
184            VrfOperation::Prove => write!(f, "PROVE"),
185            VrfOperation::Verify => write!(f, "VERIFY"),
186            VrfOperation::KeyGeneration => write!(f, "KEYGEN"),
187            VrfOperation::KeyRetrieval => write!(f, "KEYGET"),
188            VrfOperation::HsmOperation => write!(f, "HSM"),
189        }
190    }
191}

193/// Structured log entry containing operation details and metadata
194///
195/// Represents a single log event with comprehensive metadata including
196/// timestamp, severity level, operation type, and optional contextual
197/// information like key identifiers and operation duration.
198///
199/// # Fields
200///
201/// - `timestamp`: When the event occurred (UTC)
202/// - `level`: Severity level of the log entry
203/// - `operation`: Type of VRF operation being logged
204/// - `message`: Human-readable description of the event
205/// - `key_id`: Optional identifier for the key involved (if applicable)
206/// - `duration_us`: Optional operation duration in microseconds
207/// - `success`: Optional boolean indicating operation outcome
208///
209/// # Serialization
210///
211/// Log entries can be serialized to multiple formats:
212/// - Text format: Human-readable for console output
213/// - JSON format: Machine-readable for log aggregation systems
214///
215/// # Examples
216///
217/// ```rust
218/// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
219///
220/// let entry = LogEntry::new(
221///     LogLevel::Info,
222///     VrfOperation::Prove,
223///     "Generated VRF proof".to_string()
224/// );
225///
226/// // Add contextual information
227/// let entry = entry
228///     .with_key_id("validator-001".to_string())
229///     .with_success(true);
230///
231/// // Export as JSON for log aggregation
232/// println!("{}", entry.to_json());
233/// ```
234#[derive(Debug)]
235pub struct LogEntry {
236    timestamp: std::time::SystemTime,
237    level: LogLevel,
238    operation: VrfOperation,
239    message: String,
240    key_id: Option<String>,
241    duration_us: Option<u64>,
242    success: Option<bool>,
243}
244
245impl LogEntry {
246    /// Creates a new log entry with the current timestamp
247    ///
248    /// Initializes a log entry with the specified severity level, operation type,
249    /// and message. The timestamp is automatically set to the current system time.
250    /// Optional metadata can be added using builder methods.
251    ///
252    /// # Arguments
253    ///
254    /// * `level` - Log severity level (Debug, Info, Warning, Error)
255    /// * `operation` - VRF operation being logged
256    /// * `message` - Descriptive message about the event
257    ///
258    /// # Examples
259    ///
260    /// ```rust
261    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
262    ///
263    /// let entry = LogEntry::new(
264    ///     LogLevel::Info,
265    ///     VrfOperation::Prove,
266    ///     "Generated VRF proof successfully".to_string()
267    /// );
268    /// ```
269    pub fn new(level: LogLevel, operation: VrfOperation, message: String) -> Self {
270        Self {
271            timestamp: std::time::SystemTime::now(),
272            level,
273            operation,
274            message,
275            key_id: None,
276            duration_us: None,
277            success: None,
278        }
279    }
280
281    /// Adds a key identifier to the log entry
282    ///
283    /// Associates a key ID with this log entry for tracking operations
284    /// on specific VRF keys. Useful for auditing and debugging key-specific
285    /// issues in multi-key deployments.
286    ///
287    /// # Arguments
288    ///
289    /// * `key_id` - Identifier for the VRF key involved in this operation
290    ///
291    /// # Examples
292    ///
293    /// ```rust
294    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
295    ///
296    /// let entry = LogEntry::new(
297    ///     LogLevel::Info,
298    ///     VrfOperation::Prove,
299    ///     "Proof generated".to_string()
300    /// ).with_key_id("validator-key-01".to_string());
301    /// ```
302    pub fn with_key_id(mut self, key_id: String) -> Self {
303        self.key_id = Some(key_id);
304        self
305    }
306
307    /// Adds operation duration to the log entry
308    ///
309    /// Records how long the operation took to complete, stored in microseconds.
310    /// Essential for performance monitoring and identifying bottlenecks in
311    /// production VRF operations.
312    ///
313    /// # Arguments
314    ///
315    /// * `duration` - Time taken to complete the operation
316    ///
317    /// # Examples
318    ///
319    /// ```rust
320    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
321    /// use std::time::{Duration, Instant};
322    ///
323    /// let start = Instant::now();
324    /// // ... perform VRF operation ...
325    /// let elapsed = start.elapsed();
326    ///
327    /// let entry = LogEntry::new(
328    ///     LogLevel::Info,
329    ///     VrfOperation::Prove,
330    ///     "Proof completed".to_string()
331    /// ).with_duration(elapsed);
332    /// ```
333    pub fn with_duration(mut self, duration: std::time::Duration) -> Self {
334        self.duration_us = Some(duration.as_micros() as u64);
335        self
336    }
337
338    /// Adds success/failure status to the log entry
339    ///
340    /// Marks the operation as successful or failed. Critical for monitoring
341    /// error rates and identifying problematic operations in production.
342    ///
343    /// # Arguments
344    ///
345    /// * `success` - `true` if operation completed successfully, `false` otherwise
346    ///
347    /// # Examples
348    ///
349    /// ```rust
350    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
351    ///
352    /// // Successful operation
353    /// let success_entry = LogEntry::new(
354    ///     LogLevel::Info,
355    ///     VrfOperation::Verify,
356    ///     "Proof verified".to_string()
357    /// ).with_success(true);
358    ///
359    /// // Failed operation
360    /// let failure_entry = LogEntry::new(
361    ///     LogLevel::Error,
362    ///     VrfOperation::Verify,
363    ///     "Invalid proof".to_string()
364    /// ).with_success(false);
365    /// ```
366    pub fn with_success(mut self, success: bool) -> Self {
367        self.success = Some(success);
368        self
369    }
370
371    /// Formats the log entry as JSON for structured logging
372    ///
373    /// Serializes the log entry to JSON format, suitable for ingestion by
374    /// log aggregation systems like ELK, Splunk, or CloudWatch. All fields
375    /// are included with appropriate types (numbers, strings, booleans).
376    ///
377    /// # Format
378    ///
379    /// ```json
380    /// {
381    ///   "timestamp": 1704067200,
382    ///   "level": "INFO",
383    ///   "operation": "PROVE",
384    ///   "message": "Generated proof",
385    ///   "key_id": "validator-01",      // optional
386    ///   "duration_us": 1234,           // optional
387    ///   "success": true                // optional
388    /// }
389    /// ```
390    ///
391    /// # Examples
392    ///
393    /// ```rust
394    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
395    ///
396    /// let entry = LogEntry::new(
397    ///     LogLevel::Info,
398    ///     VrfOperation::Prove,
399    ///     "Generated proof".to_string()
400    /// ).with_key_id("key-01".to_string())
401    ///  .with_success(true);
402    ///
403    /// // Send to log aggregator
404    /// println!("{}", entry.to_json());
405    /// ```
406    pub fn to_json(&self) -> String {
407        let timestamp = self
408            .timestamp
409            .duration_since(std::time::UNIX_EPOCH)
410            .unwrap()
411            .as_secs();
412
413        let mut json = format!(
414            "{{\"timestamp\":{},\"level\":\"{}\",\"operation\":\"{}\",\"message\":\"{}\"",
415            timestamp, self.level, self.operation, self.message
416        );
417
418        if let Some(ref key_id) = self.key_id {
419            json.push_str(&format!(",\"key_id\":\"{}\"", key_id));
420        }
421
422        if let Some(duration) = self.duration_us {
423            json.push_str(&format!(",\"duration_us\":{}", duration));
424        }
425
426        if let Some(success) = self.success {
427            json.push_str(&format!(",\"success\":{}", success));
428        }
429
430        json.push('}');
431        json
432    }
433
434    /// Formats the log entry as human-readable text
435    ///
436    /// Generates a single-line text representation suitable for console output
437    /// or traditional log files. Includes timestamp, level, operation, message,
438    /// and optional metadata in a concise format.
439    ///
440    /// # Format
441    ///
442    /// ```text
443    /// [timestamp] LEVEL OPERATION - message | key=key_id | duration=123μs | success=true
444    /// ```
445    ///
446    /// # Examples
447    ///
448    /// ```rust
449    /// use cardano_vrf::{LogEntry, LogLevel, VrfOperation};
450    /// use std::time::Duration;
451    ///
452    /// let entry = LogEntry::new(
453    ///     LogLevel::Info,
454    ///     VrfOperation::Verify,
455    ///     "Proof verified successfully".to_string()
456    /// ).with_key_id("validator-01".to_string())
457    ///  .with_duration(Duration::from_micros(850))
458    ///  .with_success(true);
459    ///
460    /// // Output: [1704067200] INFO VERIFY - Proof verified successfully | key=validator-01 | duration=850μs | success=true
461    /// println!("{}", entry.to_text());
462    /// ```
463    pub fn to_text(&self) -> String {
464        let timestamp = self
465            .timestamp
466            .duration_since(std::time::UNIX_EPOCH)
467            .unwrap()
468            .as_secs();
469
470        let mut text = format!(
471            "[{}] {} {} - {}",
472            timestamp, self.level, self.operation, self.message
473        );
474
475        if let Some(ref key_id) = self.key_id {
476            text.push_str(&format!(" | key={}", key_id));
477        }
478
479        if let Some(duration) = self.duration_us {
480            text.push_str(&format!(" | duration={}μs", duration));
481        }
482
483        if let Some(success) = self.success {
484            text.push_str(&format!(" | success={}", success));
485        }
486
487        text
488    }

491/// Lightweight logger for VRF operations with configurable verbosity
492///
493/// Provides structured logging capabilities for VRF operations with
494/// support for multiple severity levels and filtering. In production,
495/// integrate with established logging frameworks like `tracing`, `slog`,
496/// or `log4rs` for advanced features like log rotation and forwarding.
497///
498/// # Filtering
499///
500/// Only log entries at or above the configured minimum level are emitted.
501/// This allows runtime control of logging verbosity without code changes.
502///
503/// # Output
504///
505/// By default, logs are emitted to stderr in JSON format. This can be
506/// redirected to files or piped to log aggregation systems.
507///
508/// # Examples
509///
510/// ```rust
511/// use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
512///
513/// // Create logger with Info level (filters out Debug)
514/// let logger = VrfLogger::new(LogLevel::Info);
515///
516/// // This will be logged
517/// logger.info(VrfOperation::Prove, "Proof generated".to_string());
518///
519/// // This will be filtered out
520/// logger.debug(VrfOperation::Prove, "Internal state: ...".to_string());
521/// ```
522///
523/// # Production Integration
524///
525/// ```rust,ignore
526/// use cardano_vrf::{VrfLogger, LogLevel};
527///
528/// // Set level from environment variable
529/// let level = std::env::var("LOG_LEVEL")
530///     .unwrap_or_else(|_| "INFO".to_string());
531/// let log_level = match level.as_str() {
532///     "DEBUG" => LogLevel::Debug,
533///     "WARNING" => LogLevel::Warning,
534///     "ERROR" => LogLevel::Error,
535///     _ => LogLevel::Info,
536/// };
537///
538/// let logger = VrfLogger::new(log_level);
539/// ```
540pub struct VrfLogger {
541    min_level: LogLevel,
542}
543
544impl VrfLogger {
545    /// Creates a new logger with the specified minimum log level
546    ///
547    /// Only log entries at or above this level will be emitted. This allows
548    /// runtime configuration of logging verbosity without recompilation.
549    ///
550    /// # Arguments
551    ///
552    /// * `min_level` - Minimum severity level to output (Debug < Info < Warning < Error)
553    ///
554    /// # Examples
555    ///
556    /// ```rust
557    /// use cardano_vrf::{VrfLogger, LogLevel};
558    ///
559    /// // Production logger (Info and above)
560    /// let prod_logger = VrfLogger::new(LogLevel::Info);
561    ///
562    /// // Development logger (all levels including Debug)
563    /// let dev_logger = VrfLogger::new(LogLevel::Debug);
564    ///
565    /// // Error-only logger for critical systems
566    /// let critical_logger = VrfLogger::new(LogLevel::Error);
567    /// ```
568    pub fn new(min_level: LogLevel) -> Self {
569        Self { min_level }
570    }
571
572    /// Logs an entry if it meets the minimum level threshold
573    ///
574    /// Filters log entries based on the configured minimum level. Entries
575    /// below the threshold are silently discarded. This method is typically
576    /// called by the convenience methods (`debug`, `info`, etc.) rather than
577    /// directly.
578    ///
579    /// # Output Format
580    ///
581    /// Logs are emitted to stderr in JSON format for easy parsing by log
582    /// aggregation tools. In production, redirect to a logging framework.
583    ///
584    /// # Arguments
585    ///
586    /// * `entry` - The log entry to potentially emit
587    ///
588    /// # Examples
589    ///
590    /// ```rust
591    /// use cardano_vrf::{VrfLogger, LogEntry, LogLevel, VrfOperation};
592    ///
593    /// let logger = VrfLogger::new(LogLevel::Info);
594    ///
595    /// let entry = LogEntry::new(
596    ///     LogLevel::Info,
597    ///     VrfOperation::Prove,
598    ///     "Custom log entry".to_string()
599    /// ).with_key_id("key-123".to_string());
600    ///
601    /// logger.log(entry);
602    /// ```
603    pub fn log(&self, entry: LogEntry) {
604        if entry.level >= self.min_level {
605            // In production, this would go to a proper logging framework
606            // (e.g., tracing, slog, log4rs)
607            eprintln!("{}", entry.to_json());
608        }
609    }
610
611    /// Logs a debug-level message
612    ///
613    /// Debug messages provide detailed diagnostic information useful during
614    /// development and troubleshooting. These are typically disabled in
615    /// production environments to reduce log volume.
616    ///
617    /// Use for: Internal state dumps, detailed execution flow, development aids
618    ///
619    /// # Arguments
620    ///
621    /// * `operation` - The VRF operation being performed
622    /// * `message` - Descriptive debug message
623    ///
624    /// # Examples
625    ///
626    /// ```rust
627    /// use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
628    ///
629    /// let logger = VrfLogger::new(LogLevel::Debug);
630    /// logger.debug(
631    ///     VrfOperation::Prove,
632    ///     "Processing message hash: 0x1234...".to_string()
633    /// );
634    /// ```
635    pub fn debug(&self, operation: VrfOperation, message: String) {
636        self.log(LogEntry::new(LogLevel::Debug, operation, message));
637    }

639    /// Logs an info-level message
640    ///
641    /// Info messages record normal operational events that indicate the system
642    /// is functioning correctly. This is the default logging level for production.
643    ///
644    /// Use for: Successful operations, normal state transitions, audit trails
645    ///
646    /// # Arguments
647    ///
648    /// * `operation` - The VRF operation being performed
649    /// * `message` - Descriptive informational message
650    ///
651    /// # Examples
652    ///
653    /// ```rust
654    /// use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
655    ///
656    /// let logger = VrfLogger::new(LogLevel::Info);
657    /// logger.info(
658    ///     VrfOperation::Prove,
659    ///     "VRF proof generated successfully".to_string()
660    /// );
661    /// logger.info(
662    ///     VrfOperation::Verify,
663    ///     "Proof verification succeeded".to_string()
664    /// );
665    /// ```
666    pub fn info(&self, operation: VrfOperation, message: String) {
667        self.log(LogEntry::new(LogLevel::Info, operation, message));
668    }
669
670    /// Logs a warning-level message
671    ///
672    /// Warnings indicate potential issues that don't prevent operation but may
673    /// require attention or could lead to problems if not addressed. The system
674    /// continues to function but operation may be degraded.
675    ///
676    /// Use for: Retries, degraded performance, unusual but handled conditions,
677    /// deprecated feature usage
678    ///
679    /// # Arguments
680    ///
681    /// * `operation` - The VRF operation being performed
682    /// * `message` - Description of the warning condition
683    ///
684    /// # Examples
685    ///
686    /// ```rust
687    /// use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
688    ///
689    /// let logger = VrfLogger::new(LogLevel::Info);
690    /// logger.warning(
691    ///     VrfOperation::HsmOperation,
692    ///     "HSM connection slow, operation took 5s".to_string()
693    /// );
694    /// logger.warning(
695    ///     VrfOperation::KeyRetrieval,
696    ///     "Key cache miss, loading from HSM".to_string()
697    /// );
698    /// ```
699    pub fn warning(&self, operation: VrfOperation, message: String) {
700        self.log(LogEntry::new(LogLevel::Warning, operation, message));
701    }
702
703    /// Logs an error-level message
704    ///
705    /// Errors indicate operation failures that prevent the requested action from
706    /// completing successfully. These require immediate attention and should trigger
707    /// alerts in production monitoring systems.
708    ///
709    /// Use for: Failed operations, invalid inputs, cryptographic failures, HSM errors,
710    /// unrecoverable conditions
711    ///
712    /// # Arguments
713    ///
714    /// * `operation` - The VRF operation that failed
715    /// * `message` - Detailed error description
716    ///
717    /// # Examples
718    ///
719    /// ```rust
720    /// use cardano_vrf::{VrfLogger, LogLevel, VrfOperation};
721    ///
722    /// let logger = VrfLogger::new(LogLevel::Info);
723    /// logger.error(
724    ///     VrfOperation::Verify,
725    ///     "Invalid VRF proof: verification failed".to_string()
726    /// );
727    /// logger.error(
728    ///     VrfOperation::HsmOperation,
729    ///     "HSM connection failed: timeout after 30s".to_string()
730    /// );
731    /// ```
732    pub fn error(&self, operation: VrfOperation, message: String) {
733        self.log(LogEntry::new(LogLevel::Error, operation, message));
734    }
735}

758    use super::*;
759    use std::time::Duration;
760
761    #[test]
762    fn test_log_entry_json() {
763        let entry = LogEntry::new(
764            LogLevel::Info,
765            VrfOperation::Prove,
766            "Test message".to_string(),
767        )
768        .with_key_id("test_key".to_string())
769        .with_duration(Duration::from_micros(1234))
770        .with_success(true);
771
772        let json = entry.to_json();
773        assert!(json.contains("\"level\":\"INFO\""));
774        assert!(json.contains("\"operation\":\"PROVE\""));
775        assert!(json.contains("\"message\":\"Test message\""));
776        assert!(json.contains("\"key_id\":\"test_key\""));
777        assert!(json.contains("\"duration_us\":1234"));
778        assert!(json.contains("\"success\":true"));
779    }
780
781    #[test]
782    fn test_log_entry_text() {
783        let entry = LogEntry::new(
784            LogLevel::Error,
785            VrfOperation::Verify,
786            "Verification failed".to_string(),
787        )
788        .with_success(false);
789
790        let text = entry.to_text();
791        assert!(text.contains("ERROR"));
792        assert!(text.contains("VERIFY"));
793        assert!(text.contains("Verification failed"));
794        assert!(text.contains("success=false"));
795    }
796}
