1//! Pure Rust implementation of VRF (Verifiable Random Function) for Cardano
2//!
3//! This crate provides a 100% Rust implementation of the VRF algorithms used in Cardano,
4//! following the IETF specifications:
5//! - ECVRF-ED25519-SHA512-Elligator2 (draft-03)
6//! - ECVRF-ED25519-SHA512-TAI (draft-13 batch-compatible)
7//!
8//! All implementations are memory-safe and use constant-time operations where appropriate
9//! to prevent timing attacks.
10//!
11//! # Production Features
12//!
13//! - **HSM Integration**: PKCS#11, AWS CloudHSM, Azure Key Vault support
14//! - **Metrics**: Prometheus-compatible metrics for monitoring
15//! - **Logging**: Structured audit logging for compliance
16//! - **Cryptographic Parity**: 100% byte-for-byte compatibility with Cardano libsodium
17//!
18//! # Examples
19//!
20//! ## Using Draft-03 VRF
21//!
22//! ```rust
23//! use cardano_vrf::{VrfDraft03, VrfError};
24//!
25//! # fn main() -> Result<(), VrfError> {
26//! let seed = [0u8; 32];
27//! let (secret_key, public_key) = VrfDraft03::keypair_from_seed(&seed);
28//!
29//! let message = b"Hello, Cardano!";
30//! let proof = VrfDraft03::prove(&secret_key, message)?;
31//! let output = VrfDraft03::verify(&public_key, &proof, message)?;
32//! # Ok(())
33//! # }
34//! ```
35//!
36//! ## Using Draft-13 VRF (Batch-Compatible)
37//!
38//! ```rust
39//! use cardano_vrf::{VrfDraft13, VrfError};
40//!
41//! # fn main() -> Result<(), VrfError> {
42//! let seed = [0u8; 32];
43//! let (secret_key, public_key) = VrfDraft13::keypair_from_seed(&seed);
44//!
45//! let message = b"Batch compatible VRF";
46//! let proof = VrfDraft13::prove(&secret_key, message)?;
47//! let output = VrfDraft13::verify(&public_key, &proof, message)?;
48//! # Ok(())
49//! # }
50//! ```
51//!
52//! ## Debug logging
53//!
54//! The crate ships completely silent cryptographic primitives by default. When
55//! troubleshooting interoperability issues you can enable feature `vrf-debug`
56//! and set the environment variable `CARDANO_VRF_DEBUG=1` to surface detailed
57//! diagnostics from the Cardano compatibility layer (including Elligator and
58//! hash-to-curve internals). Both the feature flag and environment variable are
59//! required to emit logs so production builds remain noise-free.

85/// Error types returned by VRF operations
86///
87/// These errors represent the various failure modes that can occur during
88/// VRF proof generation, verification, and related cryptographic operations.
89///
90/// # Security Considerations
91///
92/// Error types are deliberately generic to avoid leaking information about
93/// why a particular operation failed. Detailed error information is only
94/// available via debug logging when the `vrf-debug` feature is enabled.
95#[derive(Debug, thiserror::Error, Clone, PartialEq, Eq)]
96pub enum VrfError {
97    /// The provided VRF proof is malformed or invalid
98    ///
99    /// This can occur if:
100    /// - The proof bytes cannot be parsed (wrong length, invalid encoding)
101    /// - The proof components (Gamma, c, s) are not valid curve points/scalars
102    /// - The proof was corrupted or tampered with
103    #[error("Invalid VRF proof")]
104    InvalidProof,
105
106    /// The public key is malformed or not a valid Ed25519 point
107    ///
108    /// This occurs when the 32-byte public key cannot be decompressed to
109    /// a valid point on the Edwards curve, or represents a point with
110    /// invalid properties (e.g., not in the correct subgroup).
111    #[error("Invalid public key")]
112    InvalidPublicKey,
113
114    /// The secret key is malformed or has an invalid format
115    ///
116    /// For VRF operations, secret keys are 64 bytes (32-byte seed + 32-byte public key).
117    /// This error indicates the key doesn't meet these requirements.
118    #[error("Invalid secret key")]
119    InvalidSecretKey,
120
121    /// Failed to decode bytes as an Edwards curve point
122    ///
123    /// This occurs during hash-to-curve or proof parsing when bytes
124    /// cannot be decompressed to a valid Edwards point.
125    #[error("Invalid point encoding")]
126    InvalidPoint,
127
128    /// Failed to decode bytes as a valid scalar value
129    ///
130    /// Scalars must be in the range [0, L) where L is the order of
131    /// the Ed25519 group. This error indicates invalid encoding.
132    #[error("Invalid scalar encoding")]
133    InvalidScalar,
134
135    /// VRF proof verification failed
136    ///
137    /// The proof is well-formed but the verification equation does not hold.
138    /// This means either:
139    /// - The proof was not generated with the claimed public key
140    /// - The proof was generated for a different message
141    /// - The proof has been tampered with
142    ///
143    /// This is the expected error for invalid/forged proofs.
144    #[error("VRF verification failed")]
145    VerificationFailed,
146
147    /// Generic invalid input error with descriptive message
148    ///
149    /// Used for various input validation failures such as incorrect
150    /// buffer sizes, invalid parameters, or HSM-specific errors.
151    #[error("Invalid input: {0}")]
152    InvalidInput(String),
153}

155/// Result type alias for VRF operations
156///
157/// All VRF functions return this type, which is either a successful
158/// value of type `T` or a [`VrfError`].
159///
160/// # Examples
161///
162/// ```rust
163/// use cardano_vrf::{VrfResult, VrfError};
164///
165/// fn validate_proof() -> VrfResult<bool> {
166///     // ... validation logic
167///     Ok(true)
168/// }
169/// ```
170pub type VrfResult<T> = Result<T, VrfError>;
