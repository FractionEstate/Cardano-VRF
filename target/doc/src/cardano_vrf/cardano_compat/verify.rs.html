1//! VRF proof verification
2//!
3//! This module implements VRF proof verification matching Cardano's libsodium
4//! implementation byte-for-byte.
5

11use sha2::{Digest, Sha512};

13use super::point::{cardano_clear_cofactor, cardano_hash_to_curve};

17/// Verify VRF proof using Cardano-compatible method
18///
19/// Verifies a VRF proof and returns the VRF output if valid.
20///
21/// # Arguments
22///
23/// * `public_key` - 32-byte Ed25519 public key
24/// * `proof` - 80-byte VRF proof (Gamma || c || s)
25/// * `message` - Message that was signed
26///
27/// # Returns
28///
29/// 64-byte VRF output if proof is valid
30///
31/// # Algorithm
32///
33/// 1. Parse proof components: Gamma, c, s
34/// 2. Compute H = hash_to_curve(suite || 0x01 || pk || msg)
35/// 3. Verify equation: s*B = k*B + c*Y where k*B = s*B - c*Y
36/// 4. Verify equation: s*H = k*H + c*Gamma where k*H = s*H - c*Gamma
37/// 5. Recompute challenge c' = hash(suite || 0x02 || H || Gamma || k*B || k*H)
38/// 6. Verify c' == c
39/// 7. Compute output = hash(suite || 0x03 || Gamma)
40///
41/// # Errors
42///
43/// Returns error if proof is invalid, point decompression fails, or hash-to-curve fails
44pub fn cardano_vrf_verify(
45    public_key: &[u8; 32],
46    proof: &[u8; 80],
47    message: &[u8],
48) -> VrfResult<[u8; 64]> {
49    // Step 1: Parse proof components
50    let gamma_bytes: [u8; 32] = proof[0..32]
51        .try_into()
52        .expect("VRF proof gamma segment must be 32 bytes");
53    let c_bytes_short: [u8; 16] = proof[32..48]
54        .try_into()
55        .expect("VRF proof challenge segment must be 16 bytes");
56    let s_bytes: [u8; 32] = proof[48..80]
57        .try_into()
58        .expect("VRF proof scalar segment must be 32 bytes");
59
60    // Parse public key
61    let y_point = CompressedEdwardsY(*public_key)
62        .decompress()
63        .ok_or(VrfError::InvalidPublicKey)?;
64
65    // Parse Gamma
66    let gamma = CompressedEdwardsY(gamma_bytes)
67        .decompress()
68        .ok_or(VrfError::InvalidProof)?;
69
70    #[cfg(feature = "vrf-debug")]
71    {
72        eprintln!("Verify gamma parsing:");
73        eprintln!("  gamma_bytes (first 8): {:02x?}", &gamma_bytes[0..8]);
74        eprintln!("  gamma is_torsion_free: {}", gamma.is_torsion_free());
75    } // Parse s
76    let s = Scalar::from_bytes_mod_order(s_bytes);
77
78    // Expand c to 32 bytes
79    let mut c_bytes = [0u8; 32];
80    c_bytes[0..16].copy_from_slice(&c_bytes_short);
81    let c = Scalar::from_bytes_mod_order(c_bytes);
82
83    #[cfg(feature = "vrf-debug")]
84    {
85        eprintln!("Verify scalars:");
86        eprintln!("  s_bytes (first 8): {:02x?}", &s_bytes[0..8]);
87        eprintln!("  c_bytes (first 8): {:02x?}", &c_bytes[0..8]);
88    }
89
90    // Step 2: Hash to curve
91    let (h_point, h_string) = cardano_hash_to_curve(public_key, message)?;
92
93    #[cfg(feature = "vrf-debug")]
94    {
95        eprintln!("Verify h_point:");
96        eprintln!(
97            "  h_point bytes (first 8): {:02x?}",
98            &point_to_bytes(&h_point)[0..8]
99        );
100    }
101
102    // Step 3 & 4: Verify equations using batch scalar multiplication
103    // CRITICAL: Use batch scalar multiplication for cryptographic accuracy.
104    // This computes s*P + (-c)*Q atomically, avoiding intermediate point
105    // compression/decompression that can introduce subtle differences.
106    // This matches Cardano's libsodium reference implementation exactly.
107    let neg_c = -c;
108
109    // Compute k*B = s*B + (-c)*Y using individual scalar multiplications
110    // Note: curve25519-dalek v4 removed vartime_multiscalar_mul, we compute separately
111    let k_b = (ED25519_BASEPOINT_POINT * s) + (y_point * neg_c);
112
113    // Compute k*H = s*H + (-c)*Gamma
114    let s_h = h_point * s;
115    let c_gamma = gamma * neg_c;
116    let k_h = s_h + c_gamma;
117
118    #[cfg(feature = "vrf-debug")]
119    {
120        eprintln!("\nVerify k*H computation:");
121        eprintln!("  s scalar (first 8): {:02x?}", &s.to_bytes()[0..8]);
122        eprintln!("  c scalar (first 8): {:02x?}", &c.to_bytes()[0..8]);
123        eprintln!("  neg_c scalar (first 8): {:02x?}", &neg_c.to_bytes()[0..8]);
124        eprintln!("  c + neg_c is zero: {}", (c + neg_c) == Scalar::ZERO);
125        eprintln!(
126            "  h_point (first 8): {:02x?}",
127            &point_to_bytes(&h_point)[0..8]
128        );
129        eprintln!("  gamma (first 8): {:02x?}", &point_to_bytes(&gamma)[0..8]);
130        eprintln!("  s*H (first 8): {:02x?}", &point_to_bytes(&s_h)[0..8]);
131        eprintln!(
132            "  gamma*neg_c (first 8): {:02x?}",
133            &point_to_bytes(&c_gamma)[0..8]
134        );
135        eprintln!(
136            "  k*H = s*H + gamma*neg_c (first 8): {:02x?}",
137            &point_to_bytes(&k_h)[0..8]
138        );
139
140        // Also compute c*gamma for verification
141        let pos_c_gamma = gamma * c;
142        let neg_pos_c_gamma = -pos_c_gamma;
143
144        // Debug: test if the issue is with this specific gamma/c combination
145        let test_pt = ED25519_BASEPOINT_POINT;
146        let test_c_pt = test_pt * c;
147        let test_negc_pt = test_pt * neg_c;
148        let neg_test_c_pt = -test_c_pt;
149        eprintln!(
150            "  DEBUG: basepoint * neg_c == -(basepoint * c): {}",
151            test_negc_pt == neg_test_c_pt
152        );
153        eprintln!(
154            "  DEBUG: gamma is_torsion_free: {}",
155            gamma.is_torsion_free()
156        );
157
158        // Test with gamma directly multiplied by small scalars
159        let gamma_times_2 = gamma * Scalar::from(2u64);
160        let gamma_plus_gamma = gamma + gamma;
161        eprintln!(
162            "  DEBUG: gamma*2 == gamma+gamma: {}",
163            gamma_times_2 == gamma_plus_gamma
164        );
165
166        eprintln!(
167            "  gamma*c (for reference) (first 8): {:02x?}",
168            &point_to_bytes(&pos_c_gamma)[0..8]
169        );
170        eprintln!(
171            "  -(gamma*c) (first 8): {:02x?}",
172            &point_to_bytes(&neg_pos_c_gamma)[0..8]
173        );
174        eprintln!(
175            "  gamma*neg_c (first 8): {:02x?}",
176            &point_to_bytes(&c_gamma)[0..8]
177        );
178        eprintln!(
179            "  gamma*neg_c (before compress) == -(gamma*c): {}",
180            c_gamma == neg_pos_c_gamma
181        );
182        eprintln!(
183            "  gamma*neg_c (after compress) == -(gamma*c): {}",
184            point_to_bytes(&c_gamma) == point_to_bytes(&neg_pos_c_gamma)
185        );
186        eprintln!(
187            "  (gamma*neg_c) + (gamma*c) is_identity: {}",
188            (c_gamma + pos_c_gamma).is_identity()
189        );
190        eprintln!(
191            "  c + neg_c == Scalar::ZERO: {}",
192            (c + neg_c) == Scalar::ZERO
193        );
194        eprintln!(
195            "  gamma * (c + neg_c) is_identity: {}",
196            (gamma * (c + neg_c)).is_identity()
197        );
198        eprintln!(
199            "  gamma * Scalar::ZERO is_identity: {}",
200            (gamma * Scalar::ZERO).is_identity()
201        );
202        eprintln!("  Expected: s*H - c*gamma should equal k*H from proof");
203    }
204    let k_b_bytes = point_to_bytes(&k_b);
205    let k_h_bytes = point_to_bytes(&k_h);
206
207    #[cfg(feature = "vrf-debug")]
208    {
209        eprintln!("Verify using batch scalar multiplication:");
210        eprintln!("  s scalar (first 8): {:02x?}", &s.to_bytes()[0..8]);
211        eprintln!("  c scalar (first 8): {:02x?}", &c.to_bytes()[0..8]);
212        eprintln!("  neg_c scalar (first 8): {:02x?}", &neg_c.to_bytes()[0..8]);
213        eprintln!("  k*B result (first 8): {:02x?}", &k_b_bytes[0..8]);
214        eprintln!("  k*H result (first 8): {:02x?}", &k_h_bytes[0..8]);
215    }
216
217    // Step 5: Recompute challenge
218    let mut c_hasher = Sha512::new();
219    c_hasher.update([SUITE_DRAFT03]);
220    c_hasher.update([TWO]);
221    c_hasher.update(h_string);
222    c_hasher.update(gamma_bytes);
223    c_hasher.update(k_b_bytes);
224    c_hasher.update(k_h_bytes);
225    let c_hash = c_hasher.finalize();
226
227    // Step 6: Verify challenge matches
228    #[cfg(feature = "vrf-debug")]
229    {
230        eprintln!("Verification debug:");
231        eprintln!("  Original c: {:02x?}", &c_bytes_short[0..8]);
232        eprintln!("  Computed c: {:02x?}", &c_hash[0..8]);
233        eprintln!("  Match: {}", c_hash[0..16] == c_bytes_short[..]);
234    }
235
236    // Step 6: Verify challenge matches using constant-time comparison
237    // This is a cryptographic best practice to prevent timing attacks
238    use subtle::ConstantTimeEq;
239    let challenge_matches: bool = c_hash[0..16].ct_eq(&c_bytes_short).into();
240    if !challenge_matches {
241        return Err(VrfError::VerificationFailed);
242    }
243
244    // Step 7: Compute VRF output
245    let gamma_cleared = cardano_clear_cofactor(&gamma);
246    let mut output_hasher = Sha512::new();
247    output_hasher.update([SUITE_DRAFT03]);
248    output_hasher.update([THREE]);
249    output_hasher.update(point_to_bytes(&gamma_cleared));
250    let output_hash = output_hasher.finalize();
251
252    let mut output = [0u8; 64];
253    output.copy_from_slice(&output_hash);
254    Ok(output)
255}

261    #[test]
262    fn test_verify_compiles() {
263        let pk = [0u8; 32];
264        let proof = [0u8; 80];
265        let msg = b"test";
266
267        // Will fail with invalid proof
268        let result = cardano_vrf_verify(&pk, &proof, msg);
269        assert!(result.is_err());
270    }
271
272    #[test]
273    fn test_challenge_reconstruction() {
274        // Verify challenge bytes are correctly padded
275        let c_short = [1u8; 16];
276        let mut c_full = [0u8; 32];
277        c_full[0..16].copy_from_slice(&c_short);
278
279        assert_eq!(&c_full[0..16], &c_short);
280        assert_eq!(&c_full[16..32], &[0u8; 16]);
281    }
