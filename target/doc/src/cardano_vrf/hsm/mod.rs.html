1//! Hardware Security Module (HSM) integration for secure VRF operations
2//!
3//! This module provides a trait-based abstraction layer for delegating VRF
4//! cryptographic operations to Hardware Security Modules (HSMs). HSMs provide
5//! tamper-resistant key storage and cryptographic acceleration.
6//!
7//! # Supported Backends
8//!
9//! - **Software**: In-memory implementation for development and testing
10//! - **PKCS#11**: Industry-standard HSM interface (e.g., Thales, Utimaco)
11//! - **AWS CloudHSM**: Amazon's cloud-based HSM service
12//! - **Azure Key Vault**: Microsoft Azure's managed HSM service
13//!
14//! # Security Benefits
15//!
16//! Using an HSM for VRF operations provides:
17//! - **Key Protection**: Private keys never leave the HSM
18//! - **Audit Logging**: All cryptographic operations are logged
19//! - **Access Control**: Fine-grained permissions for key usage
20//! - **FIPS Compliance**: Certified cryptographic modules
21//!
22//! # Architecture
23//!
24//! The module uses two traits:
25//! - [`HsmVrfSigner`]: Operations requiring private keys (prove, sign)
26//! - [`HsmVrfVerifier`]: Public operations (verify)
27//!
28//! This separation allows verification to run without HSM access while
29//! keeping signing operations secure.
30//!
31//! # Examples
32//!
33//! ## Software HSM (Development)
34//!
35//! ```rust
36//! use cardano_vrf::hsm::{HsmVrfSigner, software::SoftwareVrfSigner};
37//!
38//! # fn main() -> Result<(), cardano_vrf::VrfError> {
39//! let hsm = SoftwareVrfSigner::new("/tmp/vrf-keys".to_string())?;
40//!
41//! // Generate keypair
42//! let public_key = hsm.generate_keypair("my-vrf-key")?;
43//!
44//! // Generate proof
45//! let message = b"Hello, HSM!";
46//! let proof = hsm.prove("my-vrf-key", message)?;
47//! # Ok(())
48//! # }
49//! ```
50//!
51//! ## Production Deployment
52//!
53//! In production, replace `SoftwareHsm` with a hardware-backed implementation:
54//!
55//! ```ignore
56//! use cardano_vrf::hsm::pkcs11::Pkcs11Hsm;
57//!
58//! let hsm = Pkcs11Hsm::new("/usr/lib/pkcs11/libhsm.so", "slot0")?;
59//! let proof = hsm.prove("production-vrf-key", message)?;
60//! ```

70/// Trait for VRF signing operations delegated to an HSM
71///
72/// Implementors of this trait handle VRF proof generation using private keys
73/// stored securely in an HSM. The private key never leaves the HSM hardware.
74///
75/// # Thread Safety
76///
77/// Implementations must be `Send + Sync` to support concurrent proof generation
78/// in multi-threaded applications. HSM vendors typically handle internal locking.
79///
80/// # Security Considerations
81///
82/// - **Key Isolation**: Private keys must never be exportable
83/// - **Audit Logging**: All operations should be logged by the HSM
84/// - **Rate Limiting**: Consider implementing request throttling
85/// - **Access Control**: Use HSM-native access controls for key_id
86pub trait HsmVrfSigner: Send + Sync {
87    /// Generate a VRF proof using an HSM-stored private key
88    ///
89    /// This is the core VRF operation. The HSM uses the stored private key
90    /// to generate a proof that the VRF output was computed correctly for
91    /// the given message.
92    ///
93    /// # Arguments
94    ///
95    /// * `key_id` - Identifier for the VRF signing key in the HSM (e.g., key alias, slot ID)
96    /// * `message` - Arbitrary message bytes to prove
97    ///
98    /// # Returns
99    ///
100    /// VRF proof bytes (80 bytes for Draft-03, 128 bytes for Draft-13)
101    ///
102    /// # Errors
103    ///
104    /// - [`VrfError::InvalidInput`] if key_id is not found or inaccessible
105    /// - HSM-specific errors (connection failures, authentication errors)
106    ///
107    /// # Examples
108    ///
109    /// ```rust
110    /// use cardano_vrf::hsm::{HsmVrfSigner, software::SoftwareVrfSigner};
111    ///
112    /// # fn main() -> Result<(), cardano_vrf::VrfError> {
113    /// let hsm = SoftwareVrfSigner::new("/tmp/vrf-keys".to_string())?;
114    /// hsm.generate_keypair("slot-1")?;
115    ///
116    /// let proof = hsm.prove("slot-1", b"block-12345")?;
117    /// # Ok(())
118    /// # }
119    /// ```
120    fn prove(&self, key_id: &str, message: &[u8]) -> VrfResult<Vec<u8>>;
121
122    /// Retrieve the public key for an HSM-stored private key
123    ///
124    /// Public keys can be safely exported from the HSM for verification
125    /// operations that don't require the private key.
126    ///
127    /// # Arguments
128    ///
129    /// * `key_id` - Identifier for the VRF signing key
130    ///
131    /// # Returns
132    ///
133    /// 32-byte Ed25519 public key in compressed format
134    ///
135    /// # Errors
136    ///
137    /// Returns [`VrfError::InvalidInput`] if key_id is not found
138    fn get_public_key(&self, key_id: &str) -> VrfResult<[u8; 32]>;
139
140    /// Generate a new VRF keypair in the HSM
141    ///
142    /// Creates a new Ed25519 keypair with the private key stored securely
143    /// in the HSM. The private key is generated within the HSM and never
144    /// exported.
145    ///
146    /// # Arguments
147    ///
148    /// * `key_id` - Identifier to assign to the new key
149    ///
150    /// # Returns
151    ///
152    /// Public key of the newly generated keypair
153    ///
154    /// # Errors
155    ///
156    /// - [`VrfError::InvalidInput`] if key_id already exists
157    /// - HSM-specific errors (out of storage, permission denied)
158    ///
159    /// # Examples
160    ///
161    /// ```rust
162    /// use cardano_vrf::hsm::{HsmVrfSigner, software::SoftwareVrfSigner};
163    ///
164    /// # fn main() -> Result<(), cardano_vrf::VrfError> {
165    /// let hsm = SoftwareVrfSigner::new("/tmp/vrf-keys".to_string())?;
166    /// let public_key = hsm.generate_keypair("validator-001")?;
167    /// println!("Public key: {:02x?}", &public_key[0..8]);
168    /// # Ok(())
169    /// # }
170    /// ```
171    fn generate_keypair(&self, key_id: &str) -> VrfResult<[u8; 32]>;
172
173    /// Delete a key from the HSM
174    ///
175    /// **⚠️ WARNING**: This is a destructive operation that permanently
176    /// deletes the private key. There is no recovery mechanism.
177    ///
178    /// # Security
179    ///
180    /// - Ensure proper access controls prevent unauthorized deletion
181    /// - Log all deletion attempts for audit purposes
182    /// - Consider implementing a key lifecycle policy with soft-delete
183    ///
184    /// # Arguments
185    ///
186    /// * `key_id` - Identifier of the key to delete
187    ///
188    /// # Errors
189    ///
190    /// Returns [`VrfError::InvalidInput`] if key_id is not found
191    fn delete_key(&self, key_id: &str) -> VrfResult<()>;
192
193    /// List all VRF keys available in the HSM
194    ///
195    /// Returns identifiers of all VRF keys accessible with current credentials.
196    /// Useful for key discovery and management operations.
197    ///
198    /// # Returns
199    ///
200    /// Vector of key identifiers (e.g., aliases, labels)
201    ///
202    /// # Examples
203    ///
204    /// ```rust
205    /// use cardano_vrf::hsm::{HsmVrfSigner, software::SoftwareVrfSigner};
206    ///
207    /// # fn main() -> Result<(), cardano_vrf::VrfError> {
208    /// let hsm = SoftwareVrfSigner::new("/tmp/vrf-keys".to_string())?;
209    /// hsm.generate_keypair("key-1")?;
210    /// hsm.generate_keypair("key-2")?;
211    ///
212    /// let keys = hsm.list_keys()?;
213    /// assert!(keys.contains(&"key-1".to_string()));
214    /// # Ok(())
215    /// # }
216    /// ```
217    fn list_keys(&self) -> VrfResult<Vec<String>>;
218
219    /// Test HSM connectivity and authentication
220    ///
221    /// Verifies that the HSM is reachable, credentials are valid, and
222    /// basic operations can be performed. Use this for health monitoring.
223    ///
224    /// # Returns
225    ///
226    /// `Ok(())` if the HSM is healthy and accessible
227    ///
228    /// # Errors
229    ///
230    /// Various errors depending on failure mode (network, authentication, etc.)
231    fn health_check(&self) -> VrfResult<()>;
232}

234/// Trait for VRF verification operations (public-key only)
235///
236/// Verification doesn't require access to private keys, so implementations
237/// typically don't need actual HSM connectivity. This trait exists for
238/// API consistency and future extensibility.
239pub trait HsmVrfVerifier: Send + Sync {
240    /// Verify a VRF proof
241    ///
242    /// # Arguments
243    ///
244    /// * `public_key` - 32-byte Ed25519 public key
245    /// * `proof` - VRF proof bytes
246    /// * `message` - Message that was proven
247    ///
248    /// # Returns
249    ///
250    /// 64-byte VRF output if verification succeeds
251    ///
252    /// # Errors
253    ///
254    /// Returns [`VrfError::VerificationFailed`] if the proof is invalid
255    fn verify(&self, public_key: &[u8; 32], proof: &[u8], message: &[u8]) -> VrfResult<Vec<u8>>;
256}

258/// HSM provider configuration
259#[derive(Debug, Clone)]
260pub enum HsmConfig {
261    /// PKCS#11 compliant HSM
262    Pkcs11 {
263        /// Path to PKCS#11 library (e.g., `/usr/lib/libsofthsm2.so`)
264        library_path: String,
265        /// HSM slot identifier
266        slot_id: u64,
267        /// PIN for HSM access
268        pin: String,
269    },
270    /// AWS CloudHSM
271    AwsCloudHsm {
272        /// AWS CloudHSM cluster identifier
273        cluster_id: String,
274        /// HSM user name
275        user: String,
276        /// HSM user password
277        password: String,
278    },
279    /// Azure Key Vault
280    AzureKeyVault {
281        /// Azure Key Vault URL (e.g., `https://myvault.vault.azure.net`)
282        vault_url: String,
283        /// Azure AD application (client) ID
284        client_id: String,
285        /// Azure AD application client secret
286        client_secret: String,
287        /// Azure AD tenant ID
288        tenant_id: String,
289    },
290    /// Software-based implementation (for testing)
291    Software {
292        /// Path to key storage directory
293        key_storage_path: String,
294    },
295}

297/// Factory for creating HSM-backed VRF signers
298pub struct HsmFactory;
299
300impl HsmFactory {
301    /// Create an HSM VRF signer from configuration
302    pub fn create_signer(config: HsmConfig) -> VrfResult<Box<dyn HsmVrfSigner>> {
303        match config {
304            HsmConfig::Pkcs11 {
305                library_path,
306                slot_id,
307                pin,
308            } => pkcs11::Pkcs11VrfSigner::new(library_path, slot_id, pin)
309                .map(|s| Box::new(s) as Box<dyn HsmVrfSigner>),
310            HsmConfig::AwsCloudHsm {
311                cluster_id,
312                user,
313                password,
314            } => aws_cloudhsm::AwsCloudHsmVrfSigner::new(cluster_id, user, password)
315                .map(|s| Box::new(s) as Box<dyn HsmVrfSigner>),
316            HsmConfig::AzureKeyVault {
317                vault_url,
318                client_id,
319                client_secret,
320                tenant_id,
321            } => azure_keyvault::AzureKeyVaultVrfSigner::new(
322                vault_url,
323                client_id,
324                client_secret,
325                tenant_id,
326            )
327            .map(|s| Box::new(s) as Box<dyn HsmVrfSigner>),
328            HsmConfig::Software { key_storage_path } => {
329                software::SoftwareVrfSigner::new(key_storage_path)
330                    .map(|s| Box::new(s) as Box<dyn HsmVrfSigner>)
331            }
332        }
333    }
